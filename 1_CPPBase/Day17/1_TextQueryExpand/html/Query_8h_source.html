40 // abstract class acts as a base class for concrete query types; all members are private
41 class Query_base
42 {
43  friend class Query;
44 protected:
45  typedef TextQuery::line_no line_no; // used in the eval functions
46  virtual ~Query_base() { }
47 private:
48  // eval returns the QueryResult that matches this Query

50  // rep is a string representation of the query
51  virtual std::string rep() const = 0;
52 };

54 // interface class to manage the Query_base inheritance hierarchy

68  }

71  {
72  return q->rep();
73  }
74 private:
75  Query(std::shared_ptr<Query_base> query)
76  : q(query)
